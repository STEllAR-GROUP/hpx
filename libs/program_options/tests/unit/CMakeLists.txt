set(tests
  cmdline
  options_exception
  optional
  options_description
  parsers
  positional_options
  required
  split
  unicode
  unrecognized
  variable_map
  winmain
)

if(HPX_PROGRAM_OPTIONS_WITH_BOOST_PROGRAM_OPTIONS_COMPATIBILITY)
  set(tests ${tests} boost_program_options)
endif()

foreach(test ${tests})
  set(sources
      ${test}.cpp)

  source_group("Source Files" FILES ${sources})

  set(folder_name "Tests/Unit/Modules/ProgramOptions")

  # add example executable
  add_hpx_executable(${test}_test
                     SOURCES ${sources}
                     ${${test}_FLAGS}
                     EXCLUDE_FROM_ALL
                     FOLDER ${folder_name})

  add_hpx_unit_test("modules.program_options" ${test} ${${test}_PARAMETERS})
endforeach()
