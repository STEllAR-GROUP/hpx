include(HPX_Option)

if (NOT HPX_WITH_TESTS AND HPX_TOP_LEVEL)
  hpx_set_option(HPX_DEBUGGING_WITH_TESTS VALUE OFF FORCE)
  return()
endif()
if (NOT HPX_DEBUGGING_WITH_TESTS)
  message(STATUS "Tests for debugging disabled")
  return()
endif()

if (HPX_WITH_TESTS_UNIT)
  add_hpx_pseudo_target(tests.unit.modules.debugging)
  add_hpx_pseudo_dependencies(tests.unit.modules tests.unit.modules.debugging)
  add_subdirectory(unit)
endif()

if (HPX_WITH_TESTS_REGRESSIONS)
  add_hpx_pseudo_target(tests.regressions.modules.debugging)
  add_hpx_pseudo_dependencies(tests.regressions.modules tests.regressions.modules.debugging)
  add_subdirectory(regressions)
endif()

if (HPX_WITH_TESTS_BENCHMARKS)
  add_hpx_pseudo_target(tests.performance.modules.debugging)
  add_hpx_pseudo_dependencies(tests.performance.modules tests.performance.modules.debugging)
  add_subdirectory(performance)
endif()

if (HPX_WITH_TESTS_HEADERS)
  add_hpx_header_tests(
    modules.debugging
    HEADERS ${debugging_headers}
    HEADER_ROOT ${PROJECT_SOURCE_DIR}/include
    NOLIBS
    DEPENDENCIES hpx_debugging)
