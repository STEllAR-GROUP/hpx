      add_hpx_pseudo_target(tests.unit.modules.mpi)
      add_hpx_pseudo_dependencies(tests.unit.modules tests.unit.modules.mpi)
      add_subdirectory(unit)
    endif()

    if (HPX_WITH_TESTS_REGRESSIONS)
      add_hpx_pseudo_target(tests.regressions.modules.mpi)
      add_hpx_pseudo_dependencies(tests.regressions.modules tests.regressions.modules.mpi)
      add_subdirectory(regressions)
    endif()

    if (HPX_WITH_TESTS_BENCHMARKS)
      add_hpx_pseudo_target(tests.performance.modules.mpi)
      add_hpx_pseudo_dependencies(tests.performance.modules tests.performance.modules.mpi)
      add_subdirectory(performance)
    endif()

    if (HPX_WITH_TESTS_HEADERS)
      add_hpx_header_tests(
        modules.mpi
        HEADERS ${mpi_headers}
        HEADER_ROOT ${PROJECT_SOURCE_DIR}/include
        NOLIBS
        DEPENDENCIES hpx_mpi)
    endif()
endif()
