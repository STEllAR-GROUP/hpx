set(tests
    spmd_block
    task_block
    task_block_executor
    task_block_par
   )

set(task_block_FLAGS DEPENDENCIES iostreams_component)
set(task_block_executor_FLAGS DEPENDENCIES iostreams_component)
set(task_block_par_FLAGS DEPENDENCIES iostreams_component)

foreach(test ${tests})
  set(sources
      ${test}.cpp)

  set(${test}_PARAMETERS THREADS_PER_LOCALITY 4)

  source_group("Source Files" FILES ${sources})

  # add example executable
  add_hpx_executable(${test}_test
                     SOURCES ${sources}
                     ${${test}_FLAGS}
                     EXCLUDE_FROM_ALL
                     HPX_PREFIX ${HPX_BUILD_PREFIX}
                     FOLDER "Tests/Unit/ParallelBlock")

  add_hpx_unit_test("parallel_block" ${test} ${${test}_PARAMETERS})

endforeach()
