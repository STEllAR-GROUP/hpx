set(tests
    apex_action_count
   )


foreach(test ${tests})
  set(sources
      ${test}.cpp)

  source_group("Source Files" FILES ${sources})

  # add example executable
  add_hpx_executable(${test}_test
                     SOURCES ${sources}
                     ${${test}_FLAGS}
                     EXCLUDE_FROM_ALL
                     HPX_PREFIX ${HPX_BUILD_PREFIX}
                     FOLDER "Tests/Unit/Apex")

  add_hpx_unit_test("apex" ${test} ${${test}_PARAMETERS})

  # add a custom target for this example
  add_hpx_pseudo_target(tests.unit.apex.${test})

  # make pseudo-targets depend on master pseudo-target
  add_hpx_pseudo_dependencies(tests.unit.apex
                              tests.unit.apex.${test})

  # add dependencies to pseudo-target
  add_hpx_pseudo_dependencies(tests.unit.apex.${test}
                              ${test}_test)
endforeach()
